SET @V = 50

SELECT @V

/*EXECUTANDO SELECT SEM NENHUMA TABELA*/
SELECT 500

/*SELECIONANDO TODOS OS REGISTROS DA TABELA E LISTANDO TODAS AS COLUNAS*/
SELECT *
FROM TB_DML

/*FILTRANDO POR 1 COLUNA E EXIBINDO APENAS 2 COLUNAS DA TABELA*/
SELECT 
NOME, 
DTNASC
FROM TB_DML
WHERE SEXO = 'F'

/*FILTRANDO POR 2 COLUNAS*/

FROM TB_DML
WHERE DTNASC > '1999-10-25'
AND SEXO =	'F'
